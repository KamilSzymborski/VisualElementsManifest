var TestFragment = "<Application>  <VisualElements      BackgroundColor=\"1\"      ForegroundText=\"2\"      ShowNameOnSquare150x150Logo=\"3\"      Square70x70Logo=\"4\"      Square150x150Logo=\"5\" /></Application>";
            
var Properties = ManifestService.Parse(TestFragment);

var Results = new Dictionary<ManifestPropertyType, bool>
{
    {
        ManifestPropertyType.Color,
            Properties.ContainsKey(ManifestPropertyType.Color)
                &&
            Properties[ManifestPropertyType.Color].Equals("1")
    },
    {
        ManifestPropertyType.Theme,
            Properties.ContainsKey(ManifestPropertyType.Theme)
                &&
            Properties[ManifestPropertyType.Theme].Equals("2")
    },
    {
        ManifestPropertyType.LabelStatus,
            Properties.ContainsKey(ManifestPropertyType.LabelStatus)
                &&
            Properties[ManifestPropertyType.LabelStatus].Equals("3")
    },
    {
        ManifestPropertyType.SmallImagePath,
            Properties.ContainsKey(ManifestPropertyType.SmallImagePath)
                &&
            Properties[ManifestPropertyType.SmallImagePath].Equals("4")
    },
    {
        ManifestPropertyType.MediumImagePath,
            Properties.ContainsKey(ManifestPropertyType.MediumImagePath)
                &&
            Properties[ManifestPropertyType.MediumImagePath].Equals("5")
    }
};

Results
    .ToList()
    .ForEach(Result => Console.WriteLine("{0} : {1}", Result.Key.ToString(), Result.Value? "PASSED" : "NOT PASSED"));
